gcContent <- function(dnaSet) {
  freqSet <- alphabetFrequency(dnaSet, baseOnly=TRUE, as.prob=TRUE)
  gcSet <- freqSet[,c("G", "C")]
  rowSums(gcSet)
}
dnaSet <- DNAStringSet(c("AACCAA", "GCCGTCGCT"))
print(gcContent(dnaSet))
